FUNCTION add_student():
    Prompt "Enter Student ID: "
    SET student_id = user input

    Prompt "Enter Student Name: "
    SET student_name = user input

    Prompt "Enter Student Department: "
    SET student_department = user input

    Open students file in read mode
    FOR each line in the students file:
        Split line by commas
        IF student_id matches an existing ID:
            PRINT "Error: Student ID already exists."
            RETURN

    Open students file in append mode
    Write "student_id, student_name, student_department" to the file
    PRINT "Student added successfully."

FUNCTION view_students():

        PRINT "ID: student_id | Name: student_name | Department: student_department"


FUNCTION remove_student():
    Prompt "Enter Student ID to remove: "
    SET student_id = user input

    Open students file in read mode
    READ all lines into a list

    SET updated_list = []
    SET student_found = FALSE

    FOR each line in the list:
        Split line by commas
        IF student_id matches the first field:
            SET student_found = TRUE
        ELSE:
            Append line to updated_list

    IF student_found is FALSE:
        PRINT "Error: Student ID not found."
        RETURN

    Open students file in write mode
    Write all lines from updated_list to the file
    PRINT "Student removed successfully."

FUNCTION check_student_details():
    Prompt "Enter Student ID: "
    SET student_id = user input

    Open payment records file in read mode
    READ all lines into a list

    SET student_found = FALSE
    FOR each line in the list:
        Split line by commas into student_id_from_file, name, course, fees, payment_status
        IF student_id_from_file matches student_id:
            PRINT "Student Details:"
            PRINT "ID: student_id_from_file"
            PRINT "Name: name"
            PRINT "Course: course"

FUNCTION validate_student_id(student_id):
    Open students file in read mode
    FOR each line in the file:
        Split line by commas into student_id_from_file, _
        IF student_id_from_file matches student_id:
            RETURN TRUE
    RETURN FALSE

FUNCTION read_grades_file():
    Open grades file in read mode
    SET student_ids = []
    SET student_names = []
    SET student_grades = []

    FOR each line in the file:
        Split line by commas into student_id, name, and grades
        Append student_id to student_ids
        Append name to student_names
        Append grades to student_grades

    RETURN student_ids, student_names, student_grades
